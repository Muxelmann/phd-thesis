\singlespacing

% Constants
\nomenclature{$T$}{Duration of the entire power flow simulation, here $T \in \mathbb{N}^{>0}$.}
\nomenclature{$L$}{Number of lines in the network model, here $L \in \mathbb{N}^{>0}$.}
\nomenclature{$I$}{Number of loads in the network model, here $I \in \mathbb{N}^{>0}$.}
\nomenclature{$V_n$}{Nominal voltage for a LV network, i.e. 230V P2N ($\approx \frac{400}{\sqrt{3}}\text{V}$).}
\nomenclature{$V_{ss}$}{Nominal voltage as measured at the substation transformer's secondary winding; for the IEEE LV Test Case the voltage is 254.6V P2N.}
\nomenclature{$V_l$}{Low-voltage threshold which is calculated from $V_n$,\\i.e. 245.7V P2N ($\approx V_n \times 0.94$)}
\nomenclature{$V_h$}{High-voltage threshold which is calculated from $V_n$,\\i.e. 254V P2N ($\approx V_n \times 1.1$)}

% variables
\nomenclature{$p$}{Phase number, here $p \in \{1, 2, 3\}$.}
\nomenclature{$t$}{Time step of the current simulation step, here $t \in [0, 1, \dots, \frac{T}{t_s}-1]$.}
\nomenclature{$t_s$}{Sample-period or iteration-duration used in the simulations, usually in seconds unless otherwise specified}
\nomenclature{$l$}{Line number of a given line in the network model, here $l \in [1, 2, \dots, L]$.}
\nomenclature{$i$}{Load number of a given load in the network model, here $i \in [1, 2, \dots, I]$.}

% 1D arrays
\nomenclature{$SOC(t)$}{Battery's state of charge at time $t$.}
\nomenclature{$s_{loss}(t)$}{Computed network losses at time $t$.}

% nD arrays
\nomenclature{$v_{ss,p}(t)$}{Single phase voltage measured at the substation transformer's secondary winding for phase $p$ at time $t$.}
\nomenclature{$s_{ss,p}(t)$}{Single phase apparent power measured at substation for phase $p$ at time $t$.}
\nomenclature{$s_{ESMU,p}(t)$}{ESMU's apparent power that is injected into phase $p$ at time $t$.}
\nomenclature{$\delta s_{ESMU,p}(t)$}{Change or adjustment in ESMU's apparent power that is injected into phase $p$ at time $t$.}
\nomenclature{$v_{ESMU,p}(t)$}{Single phase voltage that is measured at the ESMU's PCC for phase $p$ at time $t$.}
\nomenclature{$i_{line,l,p}(t)$}{Single phase line current that is calculated for line $l$ on phase $p$ during time $t$.}
\nomenclature{$v_{load,i,p}(t)$}{Single phase load voltage that is calculated for load $i$ on phase $p$ during time $t$.}

% functions
\nomenclature{$\zeta_\text{voltage}(v_p)$}{Voltage divergence cost for a single-phase voltage $v$ on phase $p$.}
\nomenclature{$\zeta_\text{load voltage}(v_{i,p})$}{Voltage divergence cost for a single-phase customer voltage $v$ at customer $i$, who is connected to phase $p$.}
\nomenclature{$\text{UF}(\textbf{x})$}{Unbalance Factor of a vector $\textbf{x}$.}
\nomenclature{$\zeta_\text{unbalance}(\textbf{s})$}{Unbalance cost for a three-phase power vector $\textbf{s}$ where $s_p \in \textbf{s}$.}
