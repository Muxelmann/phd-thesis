\begin{equation}
	s_{battery}(t) = 
	\begin{cases}
		\eta\text{Re}\left\{\sum_{p=1}^{P}s_{ESMU,p}(t)\right\} &\text{ if } \sum_{p=1}^{P}s_{ESMU,p}(t) \geq 0\\

	\end{cases} \forall t \text{ and } P \in \mathbb{N}
	\label{ch1:equ:battery-power-definition}
\end{equation}
